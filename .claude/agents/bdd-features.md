def check_validation_result(validation_result: Dict[str, Any], expected_result: str):
    """
    –ü—Ä–æ–≤–µ—Ä—è–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç –≤–∞–ª–∏–¥–∞—Ü–∏–∏.

@then(parsers.parse('–≤—Å–µ –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å "{expected}"'))
